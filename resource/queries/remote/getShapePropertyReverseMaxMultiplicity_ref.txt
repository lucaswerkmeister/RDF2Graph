#SELECT LIMIT ...
SELECT (max(?count) as ?max) #(min(?count) as ?min)
FROM <%4$s>
WHERE
{ 
  {
    SELECT (count(?subj) as ?count)
    {
      {
        SELECT ?subj ?target
        {
          ?subj a <%1$s> . 
          ?subj <%2$s> ?target .
          ?target a <%3$s> .     
        }
        LIMIT 100000
      }      
    }
    GROUP BY ?target 
  }
}
