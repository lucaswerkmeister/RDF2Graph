SELECT *
FROM <%4$s>
WHERE
{ 
  {
    SELECT ?subj
    {
      {
        ?subj a <%1$s> .
      }
      UNION
      { 
        ?subj rdfs:subClassOf+ <%1$s> .
      }
    }
    LIMIT 1000000 
  }
  FILTER NOT EXISTS
  {
    ?subj <%2$s> ?target .
    {
      ?target a <%3$s> .
    }
    UNION
    {
      ?target rdfs:subClassOf+ <%3$s> .
    }
  }
}
LIMIT 1
