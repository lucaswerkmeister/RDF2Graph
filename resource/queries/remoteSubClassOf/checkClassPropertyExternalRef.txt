#check whether something is an external reference or we having references to not type objects
SELECT *
FROM <%3$s>
WHERE
{ 
  {
    ?subj a <%1$s> .
  }
  UNION
  {
    ?subj rdfs:subClassOf+ <%1$s> .
  }
  ?subj <%2$s> ?target .
  ?target ?pred ?obj .
  FILTER NOT EXISTS
  {
    ?target a ?type .
  }
}
LIMIT 1
