SELECT *
FROM <%3$s>
WHERE
{
  {
    ?subj wdt:P31 <%1$s> .
  }
  UNION
  {
    ?subj wdt:P279+ <%1$s> .
  }
  FILTER NOT EXISTS
  {
    ?subj <%2$s> ?target .
    FILTER NOT EXISTS
    {
      ?target ?pred ?obj .
    }
    BIND(DATATYPE(?target) as ?dtype)
    FILTER(!bound(?dtype))
  }
}
LIMIT 1
