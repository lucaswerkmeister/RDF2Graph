SELECT DISTINCT ?refType ?xsdType
FROM <%3$s>
WHERE
{
  {
    SELECT ?subj ?target
    {
      ?subj wdt:P31 <%1$s> .
      ?subj <%2$s> ?target .
    }
    LIMIT 1000000 #virtuoso 100000
  }
  OPTIONAL
  {
    ?target wdt:P31 ?refType .
  }
  #OPTIONAL
  #{
  #  ?subj <%2$s> ?target .
  BIND(DATATYPE(?target) as ?xsdType)
  #}
}
LIMIT 1000
