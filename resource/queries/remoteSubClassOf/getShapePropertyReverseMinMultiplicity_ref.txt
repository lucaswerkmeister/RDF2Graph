SELECT *
FROM <%4$s>
WHERE
{ 
  {
    SELECT ?target
    {
      {
        ?target a <%1$s> .
      }
      UNION
      {
        ?target rdfs:subClassOf+ <%1$s> .
      }
    }
    LIMIT 1000000 
  }
  FILTER NOT EXISTS
  {
    ?subj <%2$s> ?target .
    {
      ?subj a <%3$s> .
    }
    UNION
    { 
      ?subj rdfs:subClassOf+ <%3$s> .
    }
  }
}
LIMIT 1
