SELECT *
FROM <%3$s>
WHERE
{
  {
    ?subj a <%1$s> .
  }
  UNION
  {
    ?subj rdfs:subClassOf+ <%1$s> .
  }
  FILTER NOT EXISTS
  {
    ?subj <%2$s> ?target .
    FILTER NOT EXISTS
    {
      ?target ?pred ?obj .
    }
    BIND(DATATYPE(?target) as ?dtype)
    FILTER(!bound(?dtype))
  }
}
LIMIT 1
