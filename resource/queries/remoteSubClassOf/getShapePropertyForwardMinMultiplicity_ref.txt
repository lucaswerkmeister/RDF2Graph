SELECT *
FROM <%4$s>
WHERE
{
  {
    SELECT ?subj
    {
      {
        ?subj wdt:P31 <%1$s> .
      }
      UNION
      {
        ?subj wdt:P279+ <%1$s> .
      }
    }
    LIMIT 1000000
  }
  FILTER NOT EXISTS
  {
    ?subj <%2$s> ?target .
    {
      ?target wdt:P31 <%3$s> .
    }
    UNION
    {
      ?target wdt:P279+ <%3$s> .
    }
  }
}
LIMIT 1
